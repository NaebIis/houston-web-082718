Pseudo-Code for what?

  • Understanding: If you can't explain something simply, you don't understand it. You should understand your programs.
  • Efficiency: Not planning our code leads to:
    1. Poorly Engineered, what-the-h***-is-this looking-a** code.
    2. Wasted time. It seems slower to do it at first, but this habit pays you in precious time.
    3. Frustration and despair. You can avoid them both!
  • The Jobs!! If employers ask you to write an algorithm, and you don't think it through first, you won't get the job.
  • More Jobs!! Many interviews simply consist of whiteboarding, which is basically pseudo-coding.

function helperMethod(tuplet) {
//take two elements of coll RBG
//set RBG arr
const rbg = ["R", "B", "G"]
//compare the two el to RBG arr by looping through tuplet
tuplet.forEach((char) => {
//find element
//remove from our rbg
rbg.splice(rbg.indexOf(char), 1);
});

//return rbg[0]
return rbg[0];
}
function triangle(row) {
if (row.length == 1) {
return row;
}
//row is a string of length 1 - infinity, of el R,G,B
//turn string to array
row = row.split("");
//def next-string
product = "";
//loop through the array
for (let i = 0; i < row.length - 1 ; i++) {
let el1 = row[i];
let el2 = row[i + 1];
//evaluate for next color each time
//if el1 == el2
if (el1 == el2) {
//product = el1
product += el1;
//if different,
}
else {
//product = missing color
product += helperMethod([el1, el2]);
}
}
//if current array length > 1
if (row.length > 1) {
//call self again on product
return triangle(product);
}
//else
else {
//return the last color, aka Character, derived from all previous colors
return row[0];
}
